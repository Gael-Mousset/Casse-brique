class Paddle {
/**
 * Description : Crée une raquette permet de taper la balle
 * @param {int} x Possition de la raquette en X
 * @param {int} y Possition de la raquette en Y
 * @param {int} larg Largeur de la raquette
 * @param {int} haut Hauteur de la raquette
 * @param {int} vitesse Vitesse de la raquette
 * @param {String} color Couleur de la raquette
 */ 
    constructor(x, y, larg, haut, vitesse, color) {
        this.x = x;
        this.y = y;
        this.larg = larg;
        this.haut = haut;
        this.vitesse = vitesse;
        this.color = color;
    }

/**
 * Donne la valeur de X
 * @returns {int} La possition de la raquette en X
 */
    getX() {
        return this.x;
    }
/**
 * Fixe la valeur de X
 * @param {int} x La possition de la raquette en X
 */ 
    setX(x) {
        this.x = x;
    }
/**
 * Donne la valeur de Y
 * @returns {int} La possition de la raquette en Y
 */ 
    getY() {
        return this.y;
    }
/**
 * Fixe la valeur de Y
 * @param {int} y La possition de la raquette en Y
 */ 
    setY(y) {
        this.y = y;
    }
/**
 * Donne la largeur de la raquette
 * @returns {int} La largeur de la raquette
 */ 
    getLarg() {
        return this.larg;
    }
/**
 * Fixe la largeur de la raquette
 * @param {int} larg La largeur de la raquette
 */ 
    setLarg(larg) {
        this.larg = larg;
    }
/**
 * Donne la hauteur de la raquette
 * @returns {int} La hauteur de la raquette
 */ 
    getHaut() {
        return this.haut;
    }
/**
 * Fixe la hauteur de la raquette
 * @param {any} haut La hauteur de la raquette
 */ 
    setHaut(haut) {
        this.haut = haut;
    }
/**
 * Donne la vitesse de la raquette.
 * @returns {int} La vitesse de la raquette
 */ 
    getVitesse() {
        return this.vitesse;
    }
/**
 * Fixe la vitesse de la raquette.
 * @param {int} vitesse La vitesse de la raquette
 */ 
    setVitesse(vitesse) {
        this.vitesse = vitesse;
    }
/**
 * Donne la couleur de la raquette.
 * @returns {string} Le code de la couleur de raquette
 */ 
    getColor() {
        return this.color;
    }
/**
 * Fixe la couleur de la raquette.
 * @param {string} color Le code de la couleur de raquette
 * @returns {string}
 */ 
    setColor(color) {
        this.color = color;
    }

}
