class Player {
/**
 * Description : Crée le joueur qui va jouer la partie
 * @param {String} name Nom du joueur
 * @param {int} life Nombre de vie que possède le joueur
 * @param {int} score Score du joueur
 */ 
    constructor(name, life, score) {
        this.name = name;
        this.life = life;
        this.score = score;
    }

/**
 * Donne le nom du joueur
 * @returns {string} Le nom du joueur
 */ 
    getName() {
        return this.name;
    }

/**
 * Donne le nombre de vie du joueur
 * @returns {int} Le nombre de vie que possède le joueur
 */ 
    getLife() {
        return this.life;
    }

/**
 * Donne le score actuelle du joueur
 * @returns {int} Le score du joueur
 */ 
    getScore() {
        return this.score;
    }

/**
 * Fixe le score du joueur
 * @param {int} score Le nom du joueur
 */ 
    setScore(score) {
        this.score = score;
    }

/**
 * Fixe le nombre de vie du joueur
 * @param {int} life Le nombre de vie que possède le joueur
 */ 
    setLife(life) {
        this.life = life;
    }

/**
 * Fixe le nom du joueur
 * @param {string} name  Le nom du joueur
 */ 
    setName(name) {
        this.name = name;
    }

/**
 * Fontion qui ajoute les point gagner par le joueur à sont score
 * @param {int} points Le nombre de de point donnée par une brique
 */ 
    addToScore(points) {
        this.score += points;
    }

/**
 * Fonction qui retire une vie au joueur
 */ 
    lostALife() {
        switch (this.getLife()) {
            case 1: $('#life1').attr('src', './img/life_lost.png');
                this.setLife(0)
                break;
            case 2: $('#life2').attr('src', './img/life_lost.png');
                this.setLife(1)
                break;
            case 3: $('#life3').attr('src', './img/life_lost.png');
                this.setLife(2)
                break;
        }
    }
}
