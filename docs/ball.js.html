class Ball{
    
/**
 * Description : Creation d'une balle pouvant se déplacer dans le terrain de jeux
 * @param {int} x Position en X de la balle
 * @param {int} y Position en Y de la balle
 * @param {int} radius Rayon de la balle
 * @param {int} vitesseX Vitesse en X de la balle
 * @param {int} vitesseY Vitesse en Y de la balle
 * @param {string} color Couleur de ball
 * @param {Array} matrix Matris pour gère les colistion de la ball
 */
    constructor(x, y, radius, angle, vitesseX, vitesseY, color, matrix) {
        this.x = x;
        this.y = y;
        this.radius = radius;
        this.angle = angle;
        this.vitesseX = vitesseX;
        this.vitesseY = vitesseY;
        this.color = color;
        this.matrix = matrix;
    }

/**
 * Donne la valeur de X
 * @returns {int} La position en X de la balle
 */
    getX() {
        return this.x;
    }

/**
 * Fixe la valeur de X
 * @param {int} x La position en X de la balle
 */ 
    setX(x) {
        this.x = x;
    }

/**
 * Donne la valeur Y
 * @returns {int} La position en Y de la balle
 */ 
    getY() {
        return this.y;
    }

/**
 * Fixe la valeur de Y
 * @param {int} y La position en Y de la balle
 */ 
    setY(y) {
        this.y = y;
    }

/**
 * Donne le rayon de la balle
 * @returns {int} Le rayon de la balle
 */ 
    getRadius() {
        return this.radius;
    }

/**
 * Fixe le rayon de la balle
 * @param {int} radius Le rayon de la balle
 */ 
    setRadius(radius) {
        this.radius = radius;
    }

/**
 * Donne la valeur de vitesseX
 * @returns {int} La vitesse en X de la balle
 */ 
    getVitesseX() {
        return this.vitesseX;
    }

/**
 * Fixe la valeur de vitesseX 
 * @param {int} vitesseX La vitesse en X de la balle
 */ 
    setVitesseX(vitesseX) {
        this.vitesseX = vitesseX;
    }

/**
 * Donne la valeur de vitesseY
 * @returns {int} La vitesse en Y de la balle
 */ 
    getVitesseY() {
        return this.vitesseY;
    }

/**
 * Fixe la valeur de vitesseY
 * @param {int} vitesseY La vitesse en Y de la balle
 */ 
    setVitesseY(vitesseY) {
        this.vitesseY = vitesseY;
    }

/**
 * Donne la couleur de la balle
 * @returns {string} Le code de la couleur de balle
 */ 
    getColor() {
        return this.color;
    }

/**
 * Fixe la couleur de la  balle
 * @param {string} color Le code de la couleur de balle
 */ 
    setColor(color) {
        this.color = color;
    }

/**
 * Doone la matrise de la balle sous forme d'un tableau à double entrée
 * @returns {Array} Le code matriciel de la balle
 */ 
    getMatrix() {
        return this.matrix;
    }

/**
 * Fixe la matrise de la balle sous forme d'un tableau à double entrée
 * @param {Array} matrix Le code matriciel de la balle
 */ 
    setMatrix(matrix) {
        this.matrix = matrix;
    }
}
