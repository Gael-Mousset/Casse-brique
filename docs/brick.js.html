class Brick {
/**
 * Description : Creation des brique devent être détruit durrant une partie  
 * @param {boolean} state Définis le statut de la brique
 * @param {int} x Position en X de la brqiue
 * @param {int} y Position en Y de la brique
 * @param {int} larg Largeur de la brique
 * @param {int} haut Hauteur de la brique
 * @param {int} vitesse Vitesse donner les brique vitesse
 * @param {string} color Couleur de la brique
 * @param {int} points Le nombre de points ranporter a la destrution de la brqiue
 */ 
    constructor(state, x, y, larg, haut, vitesse, color, points) {
        this.state = state;
        this.x = x;
        this.y = y;
        this.larg = larg;
        this.haut = haut;
        this.vitesse = vitesse;
        this.color = color;
        this.points = points;
    }

/**
 * Donne sont statut des brique
 * @returns {boolean} Le statut de la brique
 */ 
    getState() {
        return this.state;
    }

/**
 * Fixe le statut des brique
 * @param {boolean} state Le statut de la brique
 */ 
    setState(state) {
        this.state = state;
    }

/**
 * Donne la valeur de X
 * @returns {int} La position en X de la brqiue
 */
    getX() {
        return this.x;
    }

/**
 * Fixe la valeur de X
 * @param {int} x La position en X de la brqiue
 */ 
    setX(x) {
        this.x = x;
    }

/**
 * Donne la valeur de Y
 * @returns {int} La position en Y de la brqiue
 */ 
    getY() {
        return this.y;
    }

/**
 * Fixe la valeur de Y
 * @param {int} y La position en Y de la brqiue
 */  
    setY(y) {
        this.y = y;
    }

/**
 * Donne la largeur de la brick
 * @returns {int} La largeur de la brique
 */ 
    getLarg() {
        return this.larg;
    }

/**
 * Fixe la largeur de la brick
 * @param {int} larg La largeur de la brique
 */ 
    setLarg(larg) {
        this.larg = larg;
    }

/**
 * Donne la hauteur de la brick
 * @returns {int} La hauteur de la brique
 */ 
    getHaut() {
        return this.haut;
    }

/**
 * Fixe la hauteur de la brick
 * @param {int} haut La hauteur de la brique
 */ 
    setHaut(haut) {
        this.haut = haut;
    }

/**
 * Donne la vitesse donnée les blick vitesse
 * @returns {int} La vitesse donner les brique vitesse
 */ 
    getVitesse() {
        return this.vitesse;
    }

/**
 * Fixe la vitesse dennée par les brick vitesse
 * @param {int} vitesse La vitesse donner les brique vitesse
 */ 
    setVitesse(vitesse) {
        this.vitesse = vitesse;
    }

/**
 * Donne la couleur des brique.
 * @returns {string} Le code de la couleur de brique
 */ 
    getColor() {
        return this.color;
    }

/**
 * Fixe la couleur des brique.
 * @param {string} color Le code de la couleur de brique
 */ 
    setColor(color) {
        this.color = color;
    }

/**
 * Donne les points donnée par la brique valeur points
 * @returns {int} Le nombre de points ranporter a la destrution de la brqiue
 */ 
    getPoints() {
        return this.points;
    }

/**
 * Fixe la valeur points
 * @param {int} points Le nombre de points ranporter a la destrution de la brqiue
 */ 
    setPoints(points) {
        this.points = points;
    }

/**
 * Fonction qui dessine la brique
 */ 
    drawBrick() {
        context.beginPath();
        context.rect(this.x, this.y, this.larg, this.haut);
        context.fillStyle = this.color;
        context.fill();
        context.closePath();
    }
}
